def update_company(db: Session, company_id : int, data: CompanyUpdate):
    company = get_company(db, company_id)
    if not company:
        return None
    
    for key, value in data.model_dump(exclude_unset=True).items():
        setattr(company,key,value)

    db.commit()
    db.refresh(company)
